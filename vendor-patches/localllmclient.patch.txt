+    /// decode с переиспользованием KV общего префикса (dopishi, для автокомплита).
+    /// Сравнивает токены нового промпта с прошлым (cachedTokens), удаляет KV только после
+    /// точки расхождения (llama_memory_seq_rm) и декодирует ЛИШЬ новый хвост. При наборе
+    /// меняется конец строки -> считаем 1-2 токена вместо всего промпта.
+    func decodeReusingPrefix(text: String, addBos: Bool = false) throws(LLMError) {
+        let newTokens = [llama_token](text, addBos: addBos, special: true, vocab: vocab)
+        guard !newTokens.isEmpty else { clear(); return }
+
+        // длина общего префикса с прошлым промптом
+        var common = 0
+        let maxCommon = min(cachedTokens.count, newTokens.count)
+        while common < maxCommon, cachedTokens[common] == newTokens[common] { common += 1 }
+        // нужен минимум один НОВЫЙ токен - на нём берутся logits для генерации
+        if common == newTokens.count { common = newTokens.count - 1 }
+
+        // удаляем KV после общего префикса (старый хвост промпта + прошлая генерация)
+        if let kv = llama_get_memory(context) {
+            llama_memory_seq_rm(kv, 0, llama_pos(common), -1)
+        }
+        batch.clear()
+        logprobSum = 0
+        logprobCount = 0
+
+        // декодируем только суффикс начиная с позиции common
+        let suffix = Array(newTokens[common...])
+        let chunkSize = parameter.batch
+        for chunkStart in stride(from: 0, to: suffix.count, by: chunkSize) {
+            let chunk = suffix[chunkStart..<min(chunkStart + chunkSize, suffix.count)]
+            let chunkOffset = llama_pos(common) + llama_pos(chunkStart)
+            for (index, token) in chunk.enumerated() {
+                batch.add(id: token, pos: chunkOffset + llama_pos(index), seq_ids: [0], logits: false)
+            }
+            try decode()
+        }
+        cachedTokens = newTokens
+    }
